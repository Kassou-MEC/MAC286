public boolean isValidBST(TreeNode root) { if (root == null) return true; // Check all nodes in left subtree are less than root if (!allLessThan(root.left, root.val)) return false; // Check all nodes in right subtree are greater than root if (!allGreaterThan(root.right, root.val)) return false; return isValidBST(root.left) && isValidBST(root.right); } // Helper: returns true if all nodes in tree are less than value private boolean allLessThan(TreeNode node, int value) { if (node == null) return true; return node.val < value && allLessThan(node.left, value) && allLessThan(node.right, value); }

public class CircularQueue { private int[] arr; private int front = 0, rear = -1, count = 0; public void enqueue(int value) { rear = (rear + 1) % arr.length; arr[rear] = value; count++; } public int dequeue() { int value = arr[front]; front = (front + 1) % arr.length; count--; return value; } }
